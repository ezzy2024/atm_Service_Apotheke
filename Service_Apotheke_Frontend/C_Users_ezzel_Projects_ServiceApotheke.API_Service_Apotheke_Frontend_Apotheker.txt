    <script>
        const API_URL = "https://api.serviceapotheke.tech/api";
        const USER_ID = localStorage.getItem('userId');
        const TOKEN = localStorage.getItem('userToken') || localStorage.getItem('pharmacistToken');

        // Utils
        function showToast(msg, type = "success") {
            const container = document.getElementById('toast-container');
            const toast = document.createElement('div');
            toast.className = `toast toast-${type}`;
            toast.innerHTML = `<i class="fas ${type === 'success' ? 'fa-check-circle' : 'fa-exclamation-circle'}"></i> <span>${msg}</span>`;
            container.appendChild(toast);
            setTimeout(() => toast.classList.add('show'), 10);
            setTimeout(() => { toast.classList.remove('show'); setTimeout(() => toast.remove(), 300); }, 4000);
        }

        async function secureFetch(endpoint, options = {}) {
            const headers = { 'Authorization': `Bearer ${TOKEN}`, 'Content-Type': 'application/json', ...options.headers };
            const response = await fetch(`${API_URL}${endpoint}`, { ...options, headers });
            if (response.status === 401) logout();
            if (!response.ok) throw new Error(await response.text());
            return response.text().then(text => text ? JSON.parse(text) : {});
        }

        function logout() {
            localStorage.clear();
            window.location.replace("https://serviceapotheke.tech/login.html?logout=true");
        }

        function showSection(id) {
            document.querySelectorAll('.nav-item').forEach(el => el.classList.remove('active'));
            document.querySelectorAll('.section').forEach(el => el.classList.remove('active'));
            document.getElementById(`section-${id}`).classList.add('active');
            
            const activeNav = Array.from(document.querySelectorAll('.nav-item')).find(el => el.getAttribute('onclick').includes(id));
            if (activeNav) activeNav.classList.add('active');
            if (window.innerWidth <= 768) document.getElementById('sidebar').classList.remove('open');

            // Lazy loading logic
            if (id === 'dashboard') initDashboard();
            if (id === 'shifts') loadAvailableShifts();
            if (id === 'upcoming-shifts') loadUpcomingShifts();
            if (id === 'past-shifts') loadPastShifts();
            if (id === 'my-applications') loadMyApplications();
            if (id === 'profile') loadProfileData();
        }

        // Initialization
        document.addEventListener('DOMContentLoaded', async () => {
            document.getElementById('currentDate').textContent = new Date().toLocaleDateString('de-DE', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' });
            await loadUserData();
            initDashboard();
        });

        // API Calls & Rendering
        async function loadUserData() {
            try {
                const user = await secureFetch(`/Pharmacist/${USER_ID}`);
                document.getElementById('userFullName').textContent = user.fullName || "Apotheker";
                document.getElementById('userEmail').textContent = user.email || "";
                document.getElementById('userAvatar').textContent = user.fullName ? user.fullName.charAt(0).toUpperCase() : "P";
            } catch (e) { showToast("Benutzerdaten konnten nicht geladen werden.", "error"); }
        }

        async function initDashboard() {
            try {
                const allShifts = await secureFetch(`/Pharmacist/${USER_ID}/all-shifts`);
                const compShifts = await secureFetch(`/Pharmacist/${USER_ID}/completed-shifts`);
                const apps = await secureFetch(`/Job/JobApplication/pharmacist/${USER_ID}`);
                const upcoming = await secureFetch(`/Pharmacist/${USER_ID}/upcoming-shifts`);

                const pending = apps.filter(a => a.status === "Pending");
                const earnings = compShifts.reduce((sum, s) => sum + (s.jobPost?.salary || 0), 0);

                document.getElementById('stat-total-shifts').textContent = upcoming.length;
                document.getElementById('stat-completed-shifts').textContent = compShifts.length;
                document.getElementById('stat-pending-apps').textContent = pending.length;
                document.getElementById('stat-earnings').textContent = earnings + " â‚¬";

                const badge = document.getElementById('applications-badge');
                if (pending.length > 0) { badge.textContent = pending.length; badge.style.display = "inline-block"; } 
                else { badge.style.display = "none"; }

                renderUpcoming(upcoming, 'dashboard-upcoming-list');
            } catch(e) { document.getElementById('dashboard-upcoming-list').innerHTML = "<p style='color:red;'>Metriken nicht verfÃ¼gbar.</p>"; }
        }

        async function loadProfileData() {
            try {
                const data = await secureFetch(`/Pharmacist/${USER_ID}`);
                document.getElementById('p_fullName').value = data.fullName || '';
                document.getElementById('p_phone').value = data.phoneNumber || data.phone || '';
                document.getElementById('p_address').value = data.address || '';
                document.getElementById('p_maxDistance').value = data.maxDistanceKm || 50;
                document.getElementById('p_availableDays').value = data.availableDaysPerWeek || 5;

                const statusBadge = document.getElementById('profile-status');
                if (data.fullName && data.phoneNumber) {
                    statusBadge.textContent = "Verifiziert";

        async function loadAvailableShifts(filtered = false) {
    const container = document.getElementById('available-shifts-list');
    container.innerHTML = '<div class="loading-spinner"><i class="fas fa-spinner fa-spin"></i></div>';
    try {
        // Fetch both matches and existing applications to cross-reference
        const [matches, myApps] = await Promise.all([
            secureFetch(`/Matching/pharmacist/${USER_ID}`),
            secureFetch(`/Job/JobApplication/pharmacist/${USER_ID}`).catch(() => [])
        ]);

        if (matches.length === 0) {
            container.innerHTML = '<p style="grid-column:1/-1;">Aktuell keine Vakanzen auf dem Markt.</p>';
            return;
        }

        const appliedJobIds = myApps.map(a => a.jobPostId);
        let displayMatches = matches;

        if (filtered) {
            const fFrom = document.getElementById('filterDateFrom').value;
            const fTo = document.getElementById('filterDateTo').value;
            if (fFrom) displayMatches = displayMatches.filter(m => new Date(m.job.startDate) >= new Date(fFrom));
            if (fTo) displayMatches = displayMatches.filter(m => new Date(m.job.startDate) <= new Date(fTo));
        }

        container.innerHTML = displayMatches.map(m => {
            const s = m.job;
            const d = s.startDate ? (s.startDate.includes('-') ? new Date(s.startDate).toLocaleDateString('de-DE') : s.startDate) : '-';
            const scoreColor = m.matchScore >= 80 ? 'var(--secondary)' : (m.matchScore >= 60 ? 'var(--warning)' : 'var(--text-muted)');
            
            // Check cross-reference array
            const hasApplied = appliedJobIds.includes(s.id);
            const actionButton = hasApplied 
                ? `<button class="btn btn-outline" disabled><i class="fas fa-check"></i> Bereits beworben</button>`
                : `<button class="btn btn-primary" onclick="applyForShift(${s.id}, this)"><i class="fas fa-paper-plane"></i> Mandat anfragen</button>`;
            
            return `
                <div class="job-card">
                    <div>
                        <div class="job-header">
                            <div class="job-title">${s.pharmacy?.pharmacyName || 'Einsatzanfrage'}</div>
                            <div class="job-date" style="background: white; border: 1px solid ${scoreColor}; color: ${scoreColor};">
                                <i class="fas fa-bullseye"></i> ${m.matchScore}% Match
                            </div>
                        </div>
                        <div class="job-details">
                            <div class="detail-item"><i class="far fa-calendar-alt"></i><span>${d}</span></div>
                            <div class="detail-item"><i class="far fa-clock"></i><span>${s.startTime} - ${s.endTime}</span></div>
                            <div class="detail-item"><i class="fas fa-map-marker-alt"></i><span>${s.pharmacy?.address || 'Adresse auf Anfrage'}</span></div>
                            <div class="detail-item"><i class="fas fa-euro-sign"></i><span>${s.salary} â‚¬ / Schicht</span></div>
                        </div>
                    </div>
                    ${actionButton}
                </div>
            `;
        }).join('');
    } catch (e) { container.innerHTML = '<p style="color:var(--danger)">Routing-Engine nicht erreichbar.</p>'; }
}

        async function applyForShift(jobId, btnElement) {
    if(!confirm("MÃ¶chten Sie dieses Mandat verbindlich anfragen?")) return;
    
    // Immediate DOM Lock
    btnElement.disabled = true;
    btnElement.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Ãœbermittelt...';

    try {
        await secureFetch(`/Job/JobApplication/apply`, { 
            method: 'POST', 
            body: JSON.stringify({ pharmacistId: parseInt(USER_ID), jobPostId: jobId }) 
        });
        
        showToast("Bewerbung Ã¼bermittelt", "success");
        
        // Transform button to prevent further clicks visually
        btnElement.className = "btn btn-outline";
        btnElement.innerHTML = '<i class="fas fa-check"></i> Bereits beworben';
        
        // Update background metrics without resetting the scroll position
        initDashboard();
    } catch(e) { 
        btnElement.disabled = false;
        btnElement.innerHTML = '<i class="fas fa-paper-plane"></i> Mandat anfragen';
        showToast("Bereits beworben oder Systemfehler", "error"); 
    }
}

        async function submitTimesheet(applicationId, startDate) {
            if(!confirm("Zeiten und Kosten verbindlich zur Freigabe durch die Apotheke einreichen?")) return;
            
            const startTime = document.getElementById(`ts_start_${applicationId}`).value;
            const endTime = document.getElementById(`ts_end_${applicationId}`).value;
            const travel = document.getElementById(`ts_travel_${applicationId}`).value;
            const rate = document.getElementById(`ts_rate_${applicationId}`).value;

            const payload = {
                jobApplicationId: parseInt(applicationId),
                actualStartDate: startDate,
                actualStartTime: startTime + ":00",
                actualEndTime: endTime + ":00",
                hourlyRate: parseFloat(rate),
                travelCosts: parseFloat(travel),
                accommodationCosts: 0
            };

            try {
                await secureFetch(`/Timesheet/submit`, {
                    method: 'POST',
                    body: JSON.stringify(payload)
                });
                showToast("Zeiten erfolgreich eingereicht. Warten auf Freigabe.", "success");
                
                document.getElementById(`shift-card-${applicationId}`).innerHTML = `
                    <div style="text-align:center; padding:30px 20px; color:var(--secondary);">
                        <i class="fas fa-check-circle" style="font-size:32px; margin-bottom:15px;"></i>

                    </div>
                `;
            } catch(e) {
                showToast("Fehler bei der Ãœbermittlung der Zeiten.", "error");
            }
