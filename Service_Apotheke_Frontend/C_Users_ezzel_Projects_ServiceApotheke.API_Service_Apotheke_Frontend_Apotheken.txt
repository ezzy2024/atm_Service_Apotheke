        const API_URL = "https://api.serviceapotheke.tech/api";
        const PHARMACY_ID = localStorage.getItem('userId');
        const TOKEN = localStorage.getItem('userToken') || localStorage.getItem('pharmacyToken');

        function showToast(msg, type = "success") {
            const container = document.getElementById('toast-container');
            const toast = document.createElement('div');
            toast.className = `toast toast-${type}`;
            toast.innerHTML = `<i class="fas ${type === 'success' ? 'fa-check-circle' : 'fa-exclamation-circle'}"></i> <span>${msg}</span>`;
            container.appendChild(toast);
            setTimeout(() => toast.classList.add('show'), 10);
            setTimeout(() => { toast.classList.remove('show'); setTimeout(() => toast.remove(), 300); }, 4000);
        }

        async function secureFetch(endpoint, options = {}) {
            const headers = { 'Authorization': `Bearer ${TOKEN}`, 'Content-Type': 'application/json', ...options.headers };
            const response = await fetch(`${API_URL}${endpoint}`, { ...options, headers });
            if (response.status === 401) logout();
            if (!response.ok) throw new Error(await response.text());
            return response.text().then(text => text ? JSON.parse(text) : {});
        }

        function logout() {
            localStorage.clear();
            window.location.replace("https://serviceapotheke.tech/login.html?logout=true");
        }

        function showSection(id) {
            document.querySelectorAll('.nav-item').forEach(el => el.classList.remove('active'));
            document.querySelectorAll('.section').forEach(el => el.classList.remove('active'));
            document.getElementById(`section-${id}`).classList.add('active');
            
            const activeNav = Array.from(document.querySelectorAll('.nav-item')).find(el => el.getAttribute('onclick').includes(id));
            if (activeNav) activeNav.classList.add('active');
            if (window.innerWidth <= 768) document.getElementById('sidebar').classList.remove('open');

            if (id === 'dashboard') loadDashboardData();
            if (id === 'my-shifts') loadMyShifts();
            if (id === 'applications') loadApplications();
            if (id === 'invoices') loadInvoices();
        }

        document.addEventListener('DOMContentLoaded', async () => {
            document.getElementById('currentDate').textContent = new Date().toLocaleDateString('de-DE', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' });

            loadDashboardData();

            const urlP = new URLSearchParams(window.location.search);
            if(urlP.get('payment') === 'success') {
                showToast("Zahlung erfolgreich verarbeitet. Status wurde aktualisiert.", "success");

        function renderAppTable(list, elementId) {
            const container = document.getElementById(elementId);
            if (!list || list.length === 0) {
                container.innerHTML = '<div style="padding:40px; text-align:center; background:white; border-radius:8px; border:1px solid #e2e8f0; color:#718096;">Keine VorgÃ¤nge im Radar.</div>';
                return;
            }

            let html = `<div class="data-table-wrapper"><table class="data-table"><thead><tr><th>Kandidaten-ID</th><th>Schicht-Datum</th><th>Status</th><th>Aktion</th></tr></thead><tbody>`;
            list.forEach(a => {
                let d = 'PrÃ¼fung...';
                try {
                    const rawDate = (a.jobPost && a.jobPost.startDate) ? a.jobPost.startDate : (a.startDate ? a.startDate : null);
                    if (rawDate) { d = new Date(rawDate).toLocaleDateString('de-DE', { day: '2-digit', month: '2-digit', year: 'numeric' }); }
                } catch(e) { d = 'Datumsfehler'; }

                const sColor = a.status === 'Pending' ? 'status-pending' : (a.status === 'Accepted' ? 'status-accepted' : 'status-rejected');
                const sText = a.status === 'Pending' ? 'PrÃ¼fung ausstehend' : (a.status === 'Accepted' ? 'Zusage erteilt' : 'Absage');
                
                html += `<tr>
                    <td><strong>#APO-${a.pharmacistId || '0'}</strong></td>
                    <td>${d}</td>
                    <td><span class="status-badge ${sColor}">${sText}</span></td>
                    <td>
                        ${a.status === 'Pending' ? `
                            <div style="display:flex; gap:8px;">
                                <button class="btn btn-success" style="padding: 5px 10px; font-size: 11px; width:auto;" onclick="updateAppStatus(${a.id}, 'Accepted')"><i class="fas fa-check"></i> Zusagen</button>
                                <button class="btn btn-danger" style="padding: 5px 10px; font-size: 11px; width:auto;" onclick="updateAppStatus(${a.id}, 'Rejected')"><i class="fas fa-times"></i> Absagen</button>
                            </div>
                        ` : `<span style="color:#718096; font-size:12px;">Keine Aktion nÃ¶tig</span>`}
                    </td>
                </tr>`;
            });
            container.innerHTML = html + `</tbody></table></div>`;
        }

        async function updateAppStatus(appId, newStatus) {
            if(!confirm(`MÃ¶chten Sie diesem Bewerber verbindlich eine ${newStatus === 'Accepted' ? 'ZUSAGE' : 'ABSAGE'} erteilen?`)) return;
            try {
                await secureFetch(`/Job/JobApplication/${appId}/status`, { method: 'PUT', body: JSON.stringify({ status: newStatus }) });
                showToast(`Prozess abgeschlossen`, "success");
                loadDashboardData();
                if(document.getElementById('section-applications').classList.contains('active')) loadApplications();
            } catch (e) { showToast("Systemfehler bei der Zuweisung", "error"); }
        }
